/**

возвращает строку rgb вида rgb(3,40, 200)

@param {colour} r - The red component.
@param {colour} g - The green component.
@param {colour} b - The blue component.

*/
function rgb(r = 255, g = 255, b = 255) {
  return `rgb(${r},${g},${b})`;
}
